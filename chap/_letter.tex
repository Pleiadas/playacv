\cvsetting{\WithCoverLetter}{ % set to 1 to include application letter
	
	% _letter
	\newpage
	% Header
	\input{chap/_letterHeader}
	%
	% Opening block
	{\cvsubsectionfont \Subject}\\
	\bigskip
	%
	\input{chap/_abstract}
	\vfill
	% Dear ,
	\Greeting\ \RecipientName,\\
	%
	\smallskip
	% Body
	{\justifying \cvsetting{\WithBody}{\Body}}
	%
	\vfill
	%
	% Closer and footer
	\input{chap/_letterFooter}
	%
	\clearpage
}
